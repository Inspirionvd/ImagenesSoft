<script>
    var globalStockRaw = [];
    var globalPreciosMap = {};
    var globalMasterData = [];

    document.getElementById('fileStock').addEventListener('change', function(e) { if (e.target.files.length > 0) procesarExcelStock(e.target.files[0]); });
    document.getElementById('filePrecios').addEventListener('change', function(e) { if (e.target.files.length > 0) procesarExcelPrecios(e.target.files[0]); });

    function procesarExcelStock(archivo) {
        var reader = new FileReader();
        reader.onload = function(e) {
            var data = new Uint8Array(e.target.result);
            var workbook = XLSX.read(data, {type: 'array'});
            var hojaOrig = workbook.Sheets[workbook.SheetNames[0]];
            var rango = XLSX.utils.decode_range(hojaOrig['!ref']);
            globalStockRaw = [];
            var codigoArticuloActual = "";
            var arrayTallesActuales = [];
            for (var r = 0; r <= rango.e.r; r++) {
                var valA = obtenerValorCelda(hojaOrig, r, 0);
                var valB = obtenerValorCelda(hojaOrig, r, 1);
                if (valA !== "" && valB !== "" && obtenerValorCelda(hojaOrig, r, 2) === "") {
                    codigoArticuloActual = valA.toUpperCase();
                    for (var c = 4; c <= 18; c++) arrayTallesActuales[c] = obtenerValorCelda(hojaOrig, r, c);
                    continue;
                }
                if (codigoArticuloActual !== "") {
                    for (var c = 4; c <= 18; c++) {
                        var cant = obtenerValorCelda(hojaOrig, r, c);
                        if (cant !== "" && !isNaN(cant) && Number(cant) > 0) {
                            globalStockRaw.push({ articulo: codigoArticuloActual, codigoColor: valA, color: valB, marca: obtenerValorCelda(hojaOrig, r, 2), talle: arrayTallesActuales[c], cantidad: Number(cant) });
                        }
                    }
                }
            }
            document.getElementById('statusStock').style.display = 'block';
            verificarConsolidacion();
        };
        reader.readAsArrayBuffer(archivo);
    }

    function procesarExcelPrecios(archivo) {
        var reader = new FileReader();
        reader.onload = function(e) {
            var workbook = XLSX.read(new Uint8Array(e.target.result), {type: 'array'});
            var hoja = workbook.Sheets[workbook.SheetNames[0]];
            var rango = XLSX.utils.decode_range(hoja['!ref']);
            globalPreciosMap = {};
            for (var r = 0; r <= rango.e.r; r++) {
                var artId = obtenerValorCelda(hoja, r, 0);
                if (artId === "" || r === 0) continue;
                globalPreciosMap[artId.toUpperCase().replace(/\s+/g, '')] = {
                    precio: normalizarNumeroExcel(obtenerValorCelda(hoja, r, 3)),
                    marcaV2: obtenerValorCelda(hoja, r, 7) || "N/A",
                    precioPromo: normalizarNumeroExcel(obtenerValorCelda(hoja, r, 8)),
                    tipoVenta: obtenerValorCelda(hoja, r, 9) || "libre",
                    impresion: obtenerValorCelda(hoja, r, 10) || "N/A",
                    linea: obtenerValorCelda(hoja, r, 11) || "N/A"
                };
            }
            document.getElementById('statusPrecios').style.display = 'block';
            verificarConsolidacion();
        };
        reader.readAsArrayBuffer(archivo);
    }

    function verificarConsolidacion() {
        if (globalStockRaw.length === 0 || Object.keys(globalPreciosMap).length === 0) return;
        var mapa = {};
        globalStockRaw.forEach(function(st) {
            var key = st.articulo.toUpperCase().replace(/\s+/g, '') + "|||" + st.codigoColor;
            if (!mapa[key]) {
                var p = globalPreciosMap[st.articulo.toUpperCase().replace(/\s+/g, '')] || { precio:0, marcaV2:"N/A", precioPromo:0, tipoVenta:"libre", impresion:"N/A", linea:"N/A" };
                mapa[key] = { ...st, ...p, tallesData: [] };
            }
            mapa[key].tallesData.push({ talle: st.talle, cantidad: st.cantidad });
        });
        globalMasterData = Object.values(mapa);
        document.getElementById('panelFiltros').style.opacity = '1';
        document.getElementById('panelFiltros').style.pointerEvents = 'auto';
        inyectarSelectores();
        ejecutarFiltrosMapeados();
    }

    function inyectarSelectores() {
        var sets = { art: new Set(), col: new Set(), mrc: new Set(), mv2: new Set(), lin: new Set(), imp: new Set() };
        globalMasterData.forEach(function(i) {
            sets.art.add(i.articulo); sets.col.add(i.color); sets.mrc.add(i.marca);
            sets.mv2.add(i.marcaV2); sets.lin.add(i.linea); sets.imp.add(i.impresion);
        });
        cargarSelect('selArticulo', sets.art); cargarSelect('selColor', sets.col);
        cargarSelect('selMarca', sets.mrc); cargarSelect('selMarcaV2', sets.mv2);
        cargarSelect('selLinea', sets.lin); cargarSelect('selImpresion', sets.imp);
    }

    function cargarSelect(id, set) {
        var el = document.getElementById(id);
        el.innerHTML = '<option value="TODOS" selected>-- TODOS --</option>';
        Array.from(set).sort().forEach(v => { var o = document.createElement('option'); o.value = v; o.text = v; el.appendChild(o); });
    }

    function ejecutarFiltrosMapeados() {
        var fA = obtenerSel('selArticulo'), fC = obtenerSel('selColor'), fM = obtenerSel('selMarca'), fV = obtenerSel('selMarcaV2'), fL = obtenerSel('selLinea'), fI = obtenerSel('selImpresion');
        var res = globalMasterData.filter(i => (fA.includes("TODOS") || fA.includes(i.articulo)) && (fC.includes("TODOS") || fC.includes(i.color)) && (fM.includes("TODOS") || fM.includes(i.marca)) && (fV.includes("TODOS") || fV.includes(i.marcaV2)) && (fL.includes("TODOS") || fL.includes(i.linea)) && (fI.includes("TODOS") || fI.includes(i.impresion)));
        renderizar(res);
    }

    function renderizar(datos) {
        var cont = document.getElementById('contenedorCards');
        cont.innerHTML = "";
        document.getElementById('vistaCatalogo').style.display = 'block';
        datos.forEach(p => {
            cont.innerHTML += `<tr><td><div class="product-card"><div class="product-layout"><div class="product-img-box">${p.articulo}</div><div class="product-details"><strong>Art: ${p.articulo}</strong><br><span class="tag-badge">${p.marca}</span><span class="tag-badge tag-badge-linea">Línea: ${p.linea}</span></div><div class="product-prices">$${p.precio}</div></div></div></td></tr>`;
        });
    }

    function obtenerValorCelda(h, r, c) { var celda = h[XLSX.utils.encode_cell({r:r, c:c})]; return celda ? celda.v.toString() : ""; }
    function normalizarNumeroExcel(v) { return isNaN(v) ? 0 : Number(v); }
    function obtenerSel(id) { var s = document.getElementById(id); var res = Array.from(s.options).filter(o => o.selected).map(o => o.value); return res.includes("TODOS") ? ["TODOS"] : res; }
    function limpiarSelectoresFiltros() { ['selArticulo','selColor','selMarca','selMarcaV2','selLinea','selImpresion'].forEach(id => { var s = document.getElementById(id); for(var i=0; i<s.options.length; i++) s.options[i].selected = (s.options[i].value === "TODOS"); }); ejecutarFiltrosMapeados(); }
</script>
