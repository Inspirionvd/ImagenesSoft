<script>
    // Almacenamiento Global de Objetos Consolidados
    var globalStockRaw = [];
    var globalPreciosMap = {};
    var globalMasterData = []; 
    
    var fileStock = document.getElementById('fileStock');
    var filePrecios = document.getElementById('filePrecios');

    fileStock.addEventListener('change', function(e) {
        if (e.target.files.length > 0) procesarExcelStock(e.target.files[0]);
    });

    filePrecios.addEventListener('change', function(e) {
        if (e.target.files.length > 0) procesarExcelPrecios(e.target.files[0]);
    });

    function mostrarFeedback(texto, tipo) {
        var msg = document.getElementById('mensaje');
        msg.innerText = texto;
        msg.className = tipo;
        msg.style.display = 'block';
    }

    function ocultarFeedback() {
        document.getElementById('mensaje').style.display = 'none';
    }

    function obtenerValorCelda(hoja, r, c) {
        var direccionCelda = XLSX.utils.encode_cell({r: r, c: c});
        var celda = hoja[direccionCelda];
        return (celda && celda.v !== undefined) ? celda.v.toString().trim() : "";
    }

    // PROCESADOR DE LA MATRIZ DE STOCK POR BLOQUES
    function procesarExcelStock(archivo) {
        var reader = new FileReader();
        reader.onload = function(e) {
            try {
                var data = new Uint8Array(e.target.result);
                var workbook = XLSX.read(data, {type: 'array'});
                var hojaOrig = workbook.Sheets[workbook.SheetNames[0]];
                var rango = XLSX.utils.decode_range(hojaOrig['!ref']);
                
                globalStockRaw = [];
                var codigoArticuloActual = "";
                var arrayTallesActuales = [];
                
                for (var r = 0; r <= rango.e.r; r++) {
                    var valA = obtenerValorCelda(hojaOrig, r, 0); 
                    var valB = obtenerValorCelda(hojaOrig, r, 1); 
                    var valC = obtenerValorCelda(hojaOrig, r, 2); 
                    var valD = obtenerValorCelda(hojaOrig, r, 3); 
                    
                    if (valA === "" && valB === "" && valC === "" && valD === "") continue;
                    
                    if (valA !== "" && valB !== "" && valC === "" && valD === "") {
                        codigoArticuloActual = valA;
                        arrayTallesActuales = [];
                        for (var c = 4; c <= 18; c++) {
                            arrayTallesActuales[c] = obtenerValorCelda(hojaOrig, r, c);
                        }
                        continue;
                    }
                    
                    if (codigoArticuloActual !== "") {
                        var codigoColor = valA;
                        var nombreColor = valB;
                        var marca = valC;
                        
                        for (var c = 4; c <= 18; c++) {
                            var cantidadRaw = obtenerValorCelda(hojaOrig, r, c);
                            var talleAsociado = arrayTallesActuales[c] || "";
                            
                            if (cantidadRaw !== "" && !isNaN(cantidadRaw) && talleAsociado !== "") {
                                var cantidadNum = Number(cantidadRaw);
                                if (cantidadNum > 0) {
                                    globalStockRaw.push({
                                        articulo: codigoArticuloActual,
                                        codigoColor: codigoColor,
                                        color: nombreColor,
                                        marca: marca || "N/A",
                                        talle: talleAsociado,
                                        cantidad: cantidadNum
                                    });
                                }
                            }
                        }
                    }
                }
                document.getElementById('statusStock').style.display = 'block';
                document.getElementById('txtStock').innerText = "📄 " + archivo.name;
                verificarConsolidacion();
            } catch (err) {
                mostrarFeedback("Error al decodificar archivo de Stock: " + err.message, "error");
            }
        };
        reader.readAsArrayBuffer(archivo);
    }

    // PROCESADOR MAESTRO DE LA LISTA DE PRECIOS (EXTRAE COLUMNAS A, D, H, I, J y K)
    function procesarExcelPrecios(archivo) {
        var reader = new FileReader();
        reader.onload = function(e) {
            try {
                var data = new Uint8Array(e.target.result);
                var workbook = XLSX.read(data, {type: 'array'});
                var hojaPrecios = workbook.Sheets[workbook.SheetNames[0]]; 
                var rango = XLSX.utils.decode_range(hojaPrecios['!ref']);
                
                globalPreciosMap = {};
                
                for (var r = 0; r <= rango.e.r; r++) {
                    var artId = obtenerValorCelda(hojaPrecios, r, 0); 
                    if (artId === "" || r === 0) continue;            
                    
                    var precioReg = obtenerValorCelda(hojaPrecios, r, 3);    // Columna D (Precio)
                    var marcaV2 = obtenerValorCelda(hojaPrecios, r, 7);      // Columna H (Marca V2)
                    var precioPromo = obtenerValorCelda(hojaPrecios, r, 8);   // Columna I (Promo)
                    var tipoVenta = obtenerValorCelda(hojaPrecios, r, 9);     // Columna J (Tipo de Venta)
                    var impresion = obtenerValorCelda(hojaPrecios, r, 10);    // Columna K (Impresión)
                    
                    globalPreciosMap[artId] = {
                        precio: (precioReg !== "" && !isNaN(precioReg)) ? Number(precioReg) : 0,
                        marcaV2: marcaV2 || "N/A",
                        precioPromo: (precioPromo !== "" && !isNaN(precioPromo)) ? Number(precioPromo) : 0,
                        tipoVenta: tipoVenta || "libre",
                        impresion: impresion || "N/A"
                    };
                }
                document.getElementById('statusPrecios').style.display = 'block';
                document.getElementById('txtPrecios').innerText = "📄 " + archivo.name;
                verificarConsolidacion();
            } catch (err) {
                mostrarFeedback("Error al decodificar la lista de Precios: " + err.message, "error");
            }
        };
        reader.readAsArrayBuffer(archivo);
    }

    // VERIFICADOR Y CONSOLIDATOR DE ESTRUCTURA INTEGRAL
    function verificarConsolidacion() {
        if (globalStockRaw.length === 0 || Object.keys(globalPreciosMap).length === 0) return;
        
        ocultarFeedback();
        var mapaAgrupado = {};
        
        globalStockRaw.forEach(function(st) {
            var claveUnica = st.articulo + "|||" + st.codigoColor;
            
            if (!mapaAgrupado[claveUnica]) {
                var datosPrecio = globalPreciosMap[st.articulo] || { precio: 0, marcaV2: "N/A", precioPromo: 0, tipoVenta: "libre", impresion: "N/A" };
                
                mapaAgrupado[claveUnica] = {
                    articulo: st.articulo,
                    codigoColor: st.codigoColor,
                    color: st.color,
                    marca: st.marca,
                    marcaV2: datosPrecio.marcaV2,
                    precio: datosPrecio.precio,
                    precioPromo: datosPrecio.precioPromo,
                    tipoVenta: datosPrecio.tipoVenta,
                    impresion: datosPrecio.impresion,
                    tallesData: []
                };
            }
            
            mapaAgrupado[claveUnica].tallesData.push({
                talle: st.talle,
                cantidad: st.cantidad
            });
        });
        
        globalMasterData = Object.keys(mapaAgrupado).map(function(k) { return mapaAgrupado[k]; });
        
        inyectarSelectoresDeFiltros();
        document.getElementById('panelFiltros').style.pointerEvents = 'auto';
        document.getElementById('panelFiltros').style.opacity = '1';
        
        document.getElementById('previewTitle').style.display = 'block';
        document.getElementById('vistaCatalogo').style.display = 'block';
        
        ejecutarFiltrosMapeados();
        mostrarFeedback("¡Estructuras de datos vinculadas y consolidadas con éxito!", "success");
    }

    // CONSTRUCTOR EXCLUSIVO DE LISTADOS DESPLEGABLES DINÁMICOS
    function inyectarSelectoresDeFiltros() {
        var sets = { articulos: new Set(), colores: new Set(), marcas: new Set(), marcasV2: new Set(), impresiones: new Set() };
        
        globalMasterData.forEach(function(item) {
            sets.articulos.add(item.articulo);
            sets.colores.add(item.color);
            sets.marcas.add(item.marca);
            sets.marcasV2.add(item.marcaV2);
            sets.impresiones.add(item.impresion);
        });
        
        cargarDropdownIndividual("selArticulo", sets.articulos, "Artículos");
        cargarDropdownIndividual("selColor", sets.colores, "Colores");
        cargarDropdownIndividual("selMarca", sets.marcas, "Marcas");
        cargarDropdownIndividual("selMarcaV2", sets.marcasV2, "Marcas V2");
        cargarDropdownIndividual("selImpresion", sets.impresiones, "Impresión");
    }

    function cargarDropdownIndividual(idElemento, setValores, contexto) {
        var el = document.getElementById(idElemento);
        el.innerHTML = '<option value="TODOS" selected>-- TODOS --</option>';
        
        var ordenados = Array.from(setValores).sort();
        ordenados.forEach(function(val) {
            if(val !== "") { 
                var opt = document.createElement('option');
                opt.value = val;
                opt.innerText = (val === "N/A") ? "N/A (Sin Especificar)" : val;
                el.appendChild(opt);
            }
        });
    }

    function limpiarSelectoresFiltros() {
        var ids = ['selArticulo', 'selColor', 'selMarca', 'selMarcaV2', 'selImpresion'];
        ids.forEach(function(id) {
            var el = document.getElementById(id);
            for (var i = 0; i < el.options.length; i++) {
                el.options[i].selected = (el.options[i].value === "TODOS");
            }
        });
        ejecutarFiltrosMapeados();
    }

    // LECTURA AVANZADA DE VALORES MATRICIALES SELECCIONADOS
    function obtenerValoresSeleccionados(idSelector) {
        var el = document.getElementById(idSelector);
        var valores = [];
        for (var i = 0; i < el.options.length; i++) {
            if (el.options[i].selected) {
                valores.push(el.options[i].value);
            }
        }
        if (valores.length === 0 || valores.includes("TODOS")) {
            return ["TODOS"];
        }
        return valores;
    }

    // MOTOR LÓGICO DE FILTRADO MULTI-ITEM EN TIEMPO REAL (AHORA CON 5 FILTROS)
    function ejecutarFiltrosMapeados() {
        var fArt = obtenerValoresSeleccionados('selArticulo');
        var fCol = obtenerValoresSeleccionados('selColor');
        var fMrc = obtenerValoresSeleccionados('selMarca');
        var fMv2 = obtenerValoresSeleccionados('selMarcaV2');
        var fImp = obtenerValoresSeleccionados('selImpresion');
        
        var registrosFiltrados = globalMasterData.filter(function(item) {
            if (!fArt.includes("TODOS") && !fArt.includes(item.articulo)) return false;
            if (!fCol.includes("TODOS") && !fCol.includes(item.color)) return false;
            if (!fMrc.includes("TODOS") && !fMrc.includes(item.marca)) return false;
            if (!fMv2.includes("TODOS") && !fMv2.includes(item.marcaV2)) return false;
            if (!fImp.includes("TODOS") && !fImp.includes(item.impresion)) return false;
            return true;
        });
        
        document.getElementById('cantResultados').innerText = registrosFiltrados.length;
        renderizarEstructuraCatalogo(registrosFiltrados);
    }

    // MOTOR COMPLEMENTARIO DE ROTACIÓN EXCLUSIVA DE EXTENSIONES
    function intentarSiguienteExtension(img) {
        var extensiones = ['.jpg', '.png', '.PNG', '.jpeg', '.JPEG', '.webp', '.WEBP'];
        var idx = parseInt(img.getAttribute('data-ext-idx'));
        
        if (idx < extensiones.length) {
            var nuevaExt = extensiones[idx];
            img.setAttribute('data-ext-idx', idx + 1);
            
            var base = img.getAttribute('data-base');
            var name = img.getAttribute('data-name');
            img.src = base + encodeURIComponent(name + nuevaExt);
        } else {
            img.style.display = 'none';
            img.nextElementSibling.style.display = 'block';
        }
    }

    // RENDERIZADOR DE FICHA TÉCNICA Y GESTIÓN DE IMÁGENES POR NOMBRE
    function renderizarEstructuraCatalogo(datos) {
        var contenedor = document.getElementById('vistaCatalogo');
        contenedor.innerHTML = "";
        
        if (datos.length === 0) {
            contenedor.innerHTML = '<div style="text-align:center; padding:40px; color:#999; font-style:italic;">Ningún artículo coincide con los criterios de filtrado seleccionados.</div>';
            return;
        }
        
        var baseRuta = document.getElementById('rutaImagenes').value.trim();
        
        datos.forEach(function(prod) {
            var card = document.createElement('div');
            card.className = "product-card";
            
            var layout = document.createElement('div');
            layout.className = "product-layout";
            
            // 1. Motor de Imagen Inteligente Multi-Extensión
            var imgBox = document.createElement('div');
            imgBox.className = "product-img-box";
            
            var nombreFicheroEsperado = prod.articulo + " " + prod.color;
            
            if (baseRuta !== "") {
                var rutaLimpia = baseRuta;
                if (rutaLimpia !== "./" && !rutaLimpia.endsWith("/")) {
                    rutaLimpia += "/";
                }
                var urlCompletaImg = rutaLimpia + encodeURIComponent(nombreFicheroEsperado + ".JPG");
                
                imgBox.innerHTML = '<img src="' + urlCompletaImg + '" ' +
                                   'data-base="' + rutaLimpia + '" ' +
                                   'data-name="' + nombreFicheroEsperado + '" ' +
                                   'data-ext-idx="0" ' +
                                   'style="width:100%; height:100%; object-fit:contain; display:block;" ' +
                                   'onerror="intentarSiguienteExtension(this)">' +
                                   '<div class="img-placeholder-text" style="display:none; color:#c5221f;">Falta Imagen:<br>' + nombreFicheroEsperado + '</div>';
            } else {
                imgBox.innerHTML = '<div class="img-placeholder-text">' + nombreFicheroEsperado + '</div>';
            }
            layout.appendChild(imgBox);
            
            // 2. Información del Producto Detallada
            var details = document.createElement('div');
            details.className = "product-details";
            
            var head = document.createElement('div');
            head.className = "prod-header";
            head.innerHTML = '<h3 class="prod-title">Art: ' + prod.articulo + '</h3>' + 
                             '<p class="prod-subtitle">Color: ' + prod.color + ' <span style="color:#888;">(' + prod.codigoColor + ')</span></p>';
            details.appendChild(head);
            
            // Fila de Badges (Incluye ahora la etiqueta de Impresión)
            var tags = document.createElement('div');
            tags.className = "tags-row";
            tags.innerHTML = '<span class="tag-badge">' + (prod.marca || "N/A") + '</span>' + 
                             '<span class="tag-badge tag-badge-v2">' + (prod.marcaV2 || "N/A") + '</span>' + 
                             '<span class="tag-badge tag-badge-venta">' + prod.tipoVenta + '</span>' +
                             '<span class="tag-badge tag-badge-impresion">' + (prod.impresion || "N/A") + '</span>';
            details.appendChild(tags);
            
            // Matriz de Talles Horizontales Ordenados
            var matrix = document.createElement('div');
            matrix.className = "size-matrix";
            
            prod.tallesData.sort(function(a, b) { return a.talle.localeCompare(b.talle, undefined, {numeric: true, sensitivity: 'base'}); });
            
            prod.tallesData.forEach(function(t) {
                var sBox = document.createElement('div');
                sBox.className = "size-box";
                sBox.innerHTML = '<div class="size-name">' + t.talle + '</div>' + 
                                 '<div class="size-qty">' + t.cantidad + '</div>';
                matrix.appendChild(sBox);
            });
            details.appendChild(matrix);
            layout.appendChild(details);
            
            // 3. Columna Lateral de Gestión de Precios Financieros
            var prices = document.createElement('div');
            prices.className = "product-prices";
            
            if (prod.precioPromo > 0 && prod.precioPromo < prod.precio) {
                prices.innerHTML = '<div class="price-regular">$' + prod.precio.toLocaleString('es-AR', {minimumFractionDigits: 2}) + '</div>' + 
                                   '<div class="price-promo">$' + prod.precioPromo.toLocaleString('es-AR', {minimumFractionDigits: 2}) + '</div>' + 
                                   '<div class="promo-label">Oferta Especial</div>';
            } else if (prod.precio > 0) {
                prices.innerHTML = '<div class="price-main">$' + prod.precio.toLocaleString('es-AR', {minimumFractionDigits: 2}) + '</div>';
            } else {
                prices.innerHTML = '<div class="price-main" style="color:#b0bec5; font-size:13px;">Consultar Precio</div>';
            }
            
            layout.appendChild(prices);
            card.appendChild(layout);
            contenedor.appendChild(card);
        });
    }

    // CONTROLADOR DE COMPORTAMIENTO PREMIUM PARA SELECCIÓN MÚLTIPLE DIRECTA (SIN CTRL)
    document.addEventListener("DOMContentLoaded", function() {
        var idsFiltros = ['selArticulo', 'selColor', 'selMarca', 'selMarcaV2', 'selImpresion'];
        idsFiltros.forEach(function(id) {
            var el = document.getElementById(id);
            el.addEventListener('mousedown', function(e) {
                var target = e.target;
                if (target.tagName.toLowerCase() !== 'option') return;
                
                e.preventDefault();
                var select = this;
                var scroll = select.scrollTop;
                
                if (target.value === "TODOS") {
                    for (var i = 0; i < select.options.length; i++) {
                        select.options[i].selected = (select.options[i].value === "TODOS");
                    }
                } else {
                    target.selected = !target.selected;
                    for (var i = 0; i < select.options.length; i++) {
                        if (select.options[i].value === "TODOS") {
                            select.options[i].selected = false;
                        }
                    }
                    var comprobarSeleccion = false;
                    for (var i = 0; i < select.options.length; i++) {
                        if (select.options[i].selected) comprobarSeleccion = true;
                    }
                    if (!comprobarSeleccion) {
                        for (var i = 0; i < select.options.length; i++) {
                            if (select.options[i].value === "TODOS") select.options[i].selected = true;
                        }
                    }
                }
                
                setTimeout(function() {
                    select.scrollTop = scroll;
                }, 0);
                select.focus();
                ejecutarFiltrosMapeados();
            });
        });
    });
